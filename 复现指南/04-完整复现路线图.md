**BashTool 实现** (`src/tools/BashTool.ts`):

```typescript
import { z } from 'zod/v4';
import { exec } from 'child_process';
import { promisify } from 'util';
import type { Tool, ToolUseContext, ToolResult } from '../Tool.js';

const execAsync = promisify(exec);

const inputSchema = z.object({
  command: z.string().describe('要执行的 shell 命令'),
  timeout: z.number().optional().describe('超时时间 (ms)'),
});

export const BashTool: Tool<typeof inputSchema, { stdout: string; stderr: string; exitCode: number | null }> = {
  name: 'Bash',
  description: '执行 shell 命令并返回输出',
  inputSchema,

  isConcurrencySafe: () => false,
  isDestructive: (input) => isDestructiveCommand(input.command),

  async call(args, context): Promise<ToolResult<{ stdout: string; stderr: string; exitCode: number | null }>> {
    const { command, timeout = 30000 } = args;

    try {
      const { stdout, stderr } = await execAsync(command, {
        timeout: timeout / 1000,
        cwd: context.cwd,
        maxBuffer: 10 * 1024 * 1024, // 10MB
      });

      return {
        data: {
          stdout,
          stderr,
          exitCode: 0,
        },
      };
    } catch (error: any) {
      return {
        data: {
          stdout: error.stdout || '',
          stderr: error.stderr || error.message,
          exitCode: error.code || 1,
        },
      };
    }
  },

  mapToolResultToToolResultBlockParam(content, toolUseID) {
    const { stdout, stderr, exitCode } = content;
    const is_error = exitCode !== 0;

    return {
      type: 'tool_result',
      tool_use_id: toolUseID,
      content: [{ type: 'text', text: stdout || stderr }],
      is_error,
    };
  },
};

function isDestructiveCommand(command: string): boolean {
  const destructivePatterns = [
    /^rm\s+(-[rf]+\s+)?\//,  // rm -rf /
    /^dd\s+.*of=\/dev\//,    // dd to device
    /^mkfs/,                  // format
    /^:\(\)\{\s*:\s*&\s*\};\s*$/, // fork bomb
  ];

  return destructivePatterns.some(pattern => pattern.test(command));
}
```

**ReadTool 实现** (`src/tools/ReadTool.ts`):

```typescript
import { z } from 'zod/v4';
import { readFile } from 'fs/promises';
import type { Tool, ToolUseContext, ToolResult } from '../Tool.js';

const inputSchema = z.object({
  file_path: z.string().describe('要读取的文件路径'),
  offset: z.number().optional().describe('起始行号 (0-based)'),
  limit: z.number().optional().describe('最大读取行数'),
});

export const ReadTool: Tool<typeof inputSchema, { content: string; totalLines: number }> = {
  name: 'Read',
  description: '读取文件内容',
  inputSchema,

  isConcurrencySafe: () => true,
  isReadOnly: () => true,

  async call(args, context): Promise<ToolResult<{ content: string; totalLines: number }>> {
    const { file_path, offset = 0, limit } = args;

    // 安全检查：确保路径在工作目录内
    const fullPath = path.resolve(context.cwd, file_path);
    if (!fullPath.startsWith(context.cwd)) {
      return {
        data: { content: '', totalLines: 0 },
        newMessages: [{
          type: 'user',
          content: [{ type: 'text', text: '错误：不允许访问工作目录外的文件' }],
        }],
      };
    }

    try {
      const content = await readFile(fullPath, 'utf-8');
      const lines = content.split('\n');

      const slicedLines = limit
        ? lines.slice(offset, offset + limit)
        : lines.slice(offset);

      return {
        data: {
          content: slicedLines.join('\n'),
          totalLines: lines.length,
        },
      };
    } catch (error: any) {
      return {
        data: {
          content: `错误：${error.message}`,
          totalLines: 0,
        },
      };
    }
  },

  mapToolResultToToolResultBlockParam(content, toolUseID) {
    return {
      type: 'tool_result',
      tool_use_id: toolUseID,
      content: [{ type: 'text', text: content.content }],
      is_error: content.content.startsWith('错误：'),
    };
  },
};
```

```typescript
    async *streamMessage(params: {

      messages: MessageCreateParams['messages'];
```

**查询循环** (`src/query.ts`):

```typescript
import type { MessageCreateParams, MessageStreamEvent } from '@anthropic-ai/sdk/resources/messages.mjs';
import type { Tool, ToolUseContext } from './Tool.js';
import type { APIClient } from './services/api.js';

interface QueryOptions {
  model: string;
  messages: Message[];
  systemPrompt?: string;
  tools: Tool[];
  maxTurns: number;
  maxTokens: number;
}

export async function* runQuery(options: QueryOptions): AsyncGenerator<QueryEvent> {
  const { model, messages, systemPrompt, tools, maxTurns, maxTokens } = options;

  let currentMessages = [...messages];
  let turnCount = 0;

  while (turnCount < maxTurns) {
    turnCount++;

    // 1. 调用 API
    const toolDefs = normalizeTools(tools);
    const apiParams: MessageCreateParams = {
      model,
      max_tokens: maxTokens,
      messages: currentMessages,
      system: systemPrompt,
      tools: toolDefs,
    };

    yield { type: 'query_start', turn: turnCount };

    // 2. 流式处理响应
    const assistantMessage: AssistantMessage = {
      type: 'assistant',
      content: [],
    };

    for await (const event of apiClient.streamMessage(apiParams)) {
      if (event.type === 'content_block_delta') {
        assistantMessage.content.push(event.delta);
        yield { type: 'content_delta', delta: event.delta };
      }
    }

    // 3. 检查是否有工具调用
    const toolUses = assistantMessage.content.filter(c => c.type === 'tool_use');

    if (toolUses.length === 0) {
      // 没有工具调用，查询完成
      yield { type: 'query_complete', message: assistantMessage };
      break;
    }

    // 4. 执行工具
    const toolResults = await executeTools(toolUses, tools, currentMessages);

    // 5. 添加工具结果到消息历史
    currentMessages.push(assistantMessage);
    currentMessages.push(...toolResults);

    yield { type: 'tools_executed', count: toolUses.length };
  }

  if (turnCount >= maxTurns) {
    yield { type: 'max_turns_reached', turns: turnCount };
  }
}

async function executeTools(
  toolUses: ToolUseBlock[],
  tools: Tool[],
  messages: Message[]
): Promise<Message[]> {
  const results: Message[] = [];

  for (const toolUse of toolUses) {
    const tool = tools.find(t => t.name === toolUse.name);
    if (!tool) continue;

    const context: ToolUseContext = {
      cwd: process.cwd(),
      permissionMode: 'default',
      canUseTool: async () => ({ behavior: 'allow' }),
    };

    try {
      const result = await tool.call(toolUse.input as any, context);
      results.push({
        type: 'user',
        content: [tool.mapToolResultToToolResultBlockParam(result.data, toolUse.id)],
      });
    } catch (error: any) {
      results.push({
        type: 'user',
        content: [{
          type: 'tool_result',
          tool_use_id: toolUse.id,
          content: [{ type: 'text', text: `错误：${error.message}` }],
          is_error: true,
        }],
      });
    }
  }

  return results;
}
```

**权限系统** (`src/utils/permissions.ts`):

```typescript
export type PermissionMode = 'default' | 'auto' | 'plan' | 'bypass';

export interface PermissionRule {
  type: 'alwaysAllow' | 'alwaysDeny' | 'alwaysAsk';
  tool?: string;
  pattern?: string;
  path?: string;
}

export interface PermissionResult {
  behavior: 'allow' | 'deny' | 'ask';
  reason?: string;
}

export async function checkPermission(
  tool: Tool,
  input: unknown,
  mode: PermissionMode,
  rules: PermissionRule[]
): Promise<PermissionResult> {
  // bypass 模式：总是允许
  if (mode === 'bypass') {
    return { behavior: 'allow' };
  }

  // auto 模式：只允许只读操作
  if (mode === 'auto') {
    if (tool.isReadOnly?.(input as any)) {
      return { behavior: 'allow' };
    }
    return { behavior: 'deny', reason: 'auto 模式下只允许只读操作' };
  }

  // 规则匹配
  for (const rule of rules) {
    if (matchesRule(tool, input, rule)) {
      return {
        behavior: rule.type === 'alwaysAllow' ? 'allow' :
                  rule.type === 'alwaysDeny' ? 'deny' : 'ask',
      };
    }
  }

  // 默认行为
  if (mode === 'plan') {
    return { behavior: 'ask' };
  }

  // default 模式：危险操作需要询问
  if (tool.isDestructive?.(input as any)) {
    return { behavior: 'ask' };
  }

  return { behavior: 'allow' };
}

function matchesRule(tool: Tool, input: unknown, rule: PermissionRule): boolean {
  if (rule.tool && rule.tool !== tool.name) return false;

  if (rule.pattern) {
    // 简单通配符匹配
    const regex = new RegExp(rule.pattern.replace('*', '.*'));
    const command = (input as any)?.command;
    if (command && !regex.test(command)) return false;
  }

  if (rule.path) {
    const filePath = (input as any)?.file_path;
    if (filePath && !filePath.startsWith(rule.path)) return false;
  }

  return true;
}
```

**基础 REPL** (`src/repl.ts`):

```typescript
import * as readline from 'readline';
import { runQuery } from './query.js';
import type { Tool } from './Tool.js';
import type { APIClient } from './services/api.js';

interface REPLOptions {
  tools: Tool[];
  apiClient: APIClient;
  model: string;
  permissionMode: string;
  maxTurns: number;
}

export async function runREPL(options: REPLOptions) {
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
  });

  const messages: Message[] = [];
  const systemPrompt = buildSystemPrompt(options.tools);

  console.log(`Claude Code Clone v0.1.0 (模型：${options.model})`);
  console.log('输入 /help 查看可用命令，输入 /quit 退出\n');

  const promptLoop = async () => {
    rl.question('> ', async (input) => {
      try {
        // 处理斜杠命令
        if (input.startsWith('/')) {
          await handleSlashCommand(input.slice(1), messages);
          promptLoop();
          return;
        }

        // 添加用户消息
        messages.push({
          type: 'user',
          content: [{ type: 'text', text: input }],
        });

        // 运行查询
        for await (const event of runQuery({
          model: options.model,
          messages,
          systemPrompt,
          tools: options.tools,
          maxTurns: options.maxTurns,
          maxTokens: 8192,
        })) {
          handleQueryEvent(event);
        }

      } catch (error: any) {
        console.error(`错误：${error.message}`);
      }

      promptLoop();
    });
  };

  promptLoop();
}

function handleQueryEvent(event: QueryEvent) {
  switch (event.type) {
    case 'query_start':
      console.log(`\n[第 ${event.turn} 轮]`);
      break;
    case 'content_delta':
      process.stdout.write(event.delta.text || '');
      break;
    case 'tools_executed':
      console.log(`\n[执行了 ${event.count} 个工具]`);
      break;
    case 'query_complete':
      console.log('\n');
      break;
  }
}
```

```typescript
// src/services/costTracker.ts
const PRICING = {
  'claude-sonnet-4-6-20260320': {
    input: 3 / 1_000_000,
    output: 15 / 1_000_000,
  },
  'claude-opus-4-6-20260320': {
    input: 15 / 1_000_000,
    output: 75 / 1_000_000,
  },
};

export class CostTracker {
  private inputTokens = 0;
  private outputTokens = 0;

  accumulateUsage(usage: { input_tokens: number; output_tokens: number }) {
    this.inputTokens += usage.input_tokens;
    this.outputTokens += usage.output_tokens;
  }

  calculateCost(model: string): number {
    const pricing = PRICING[model as keyof typeof PRICING];
    if (!pricing) return 0;

    return (
      this.inputTokens * pricing.input +
      this.outputTokens * pricing.output
    );
  }
}
```

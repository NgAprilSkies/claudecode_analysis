#### 2.1.3 管道命令分析

```typescript
export function isSearchOrReadBashCommand(command: string): {
  isSearch: boolean;
  isRead: boolean;
  isList: boolean;
} {
  const parts = splitCommandWithOperators(command);

  for (const part of parts) {
    // 跳过重定向目标
    if (part === '>' || part === '>>') continue;

    const baseCommand = part.trim().split(/\s+/)[0];

    // 跳过中性命令 (echo, printf 等)
    if (BASH_SEMANTIC_NEUTRAL_COMMANDS.has(baseCommand)) continue;

    // 管道中所有部分都必须是 search/read 才能折叠
    if (!isSearchAnd !isReadAnd !isListCommand) {
      return { isSearch: false, isRead: false, isList: false };
    }
  }

  return { isSearch, isRead, isList };
}
```

#### 2.1.4 执行流程

```typescript
async function call(input, context, canUseTool, parentMessage, onProgress) {
  const { command, timeout, background } = input;

  // 1. 命令安全性检查
  const validation = await validateInput(input, context);
  if (!validation.result) return error(validation.message);

  // 2. 权限检查
  const permission = await checkPermissions(input, context);
  if (permission.behavior !== 'allow') return denied();

  // 3. 读取命令检查 (只读模式限制)
  const readOnlyCheck = await checkReadOnlyConstraints(input, context);
  if (!readOnlyCheck.allowed) return readOnlyError();

  // 4. 执行命令
  const result = await executeCommand(command, {
    timeout,
    background,
    cwd: context.cwd,
    onProgress: (data) => onProgress?.({ data }),
  });

  // 5. 处理输出
  const { stdout, stderr, exitCode } = result;

  // 6. 大输出处理 (超过阈值则写入文件)
  if (stdout.length > MAX_RESULT_SIZE) {
    const path = await writeLargeResult(stdout);
    return { path, truncated: true };
  }

  // 7. 图像输出检测
  if (isImageOutput(stdout)) {
    return buildImageToolResult(stdout);
  }

  return { stdout, stderr, exitCode };
}
```

```typescript
// bashPermissions.ts
export function bashToolHasPermission(
  command: string,
  permissionRules: ToolPermissionRulesBySource,
  action: 'allow' | 'deny' | 'ask'
): boolean {
  const prefix = permissionRuleExtractPrefix(command);

  for (const pattern of permissionRules) {
    // 支持通配符匹配：git *, npm install
    if (matchWildcardPattern(pattern, command)) {
      return true;
    }
    // cd 命令特殊处理 (路径匹配)
    if (command.startsWith('cd ')) {
      const cdPath = command.slice(3);
      if (matchWildcardPattern(pattern, `cd ${cdPath}`)) {
        return true;
      }
    }
  }

  return false;
}

#### 2.2.2 大文件处理

// 文件读取限制
const FILE_READ_LIMITS = {
  maxTokens: 4000,      // 最大 Token 数
  maxSizeBytes: 1024 * 1024, // 1MB
};

async function readTextFile(path: string, options) {
  const { offset = 0, limit } = options;

  // 检查文件大小
  const stats = await fs.stat(path);
  if (stats.size > FILE_READ_LIMITS.maxSizeBytes) {
    return {
      error: `文件过大 (${formatBytes(stats.size)})`,
      suggestion: '使用 offset 和 limit 参数分段读取',
    };
  }

  const content = await fs.readFile(path, 'utf-8');
  const lines = content.split('\n');

  // 支持分页读取
  const slicedLines = limit
    ? lines.slice(offset, offset + limit)
    : lines.slice(offset);

  return {
    content: slicedLines.join('\n'),
    totalLines: lines.length,
    returnedLines: slicedLines.length,
    hasMore: offset + slicedLines.length < lines.length,
  };
}
```

#### 2.3.3 安全写入 (权限检查)

```typescript
async function checkPermissions(input, context) {
  const { file_path } = input;

  // 检查是否在允许的工作目录内
  const allowedDirs = getAdditionalWorkingDirectories(context);
  const isInAllowedDir = allowedDirs.some(dir =>
    file_path.startsWith(dir.path)
  );

  // 检查是否为危险操作
  const isDestructive = await checkDestructiveOperation(file_path, input);

  // 检查只读限制
  if (context.readOnlyMode) {
    return { behavior: 'deny', reason: '当前为只读模式' };
  }

  // 检查危险权限规则
  if (hasDenyRule(file_path, context.alwaysDenyRules)) {
    return { behavior: 'deny', reason: '匹配拒绝规则' };
  }

  if (isInAllowedDir && !isDestructive) {
    return { behavior: 'allow' };
  }

  return { behavior: 'ask' };
}

const inputSchema = z.object({
  pattern: z.string().describe('正则表达式模式'),
  path: z.string().optional().describe('搜索路径'),
  glob: z.string().optional().describe('文件匹配模式 (如 *.ts)'),
  case_sensitive: z.boolean().optional(),
  max_results: z.number().optional().default(100),
});

async function call(input, context) {
  const { pattern, path, glob, case_sensitive, max_results } = input;

  // 构建 ripgrep 命令
  const args = [
    '--json',           // JSON 输出
    '--max-count', String(max_results),
    '--regexp', pattern,
  ];

  if (glob) args.push('--glob', glob);
  if (case_sensitive) args.push('--case-sensitive');
  if (path) args.push(path);

  // 执行 rg
  const result = await exec('rg', args, { cwd: context.cwd });

  // 解析 JSONL 输出
  const matches = parseRipgrepOutput(result.stdout);

  return {
    matches,
    totalMatches: matches.length,
    hasMore: result.stdout.includes('"reason":"max_matches_found"'),
  };
}
```

#### 2.4.2 输出解析

```typescript
interface RipgrepMatch {
  path: string;
  line_number: number;
  column: number;
  text: string;
  submatches: Array<{ start: number; end: number }>;
}

function parseRipgrepOutput(output: string): RipgrepMatch[] {
  const matches: RipgrepMatch[] = [];

  for (const line of output.split('\n').filter(Boolean)) {
    try {
      const parsed = JSON.parse(line);

      if (parsed.type === 'match') {
        matches.push({
          path: parsed.data.path.text,
          line_number: parsed.data.line_number,
          column: parsed.data.submatches[0]?.start || 0,
          text: parsed.data.lines.text,
          submatches: parsed.data.submatches,
        });
      }
    } catch {
      // 跳过无效 JSON 行
    }
  }

  return matches;
}
```

  if (!restrictions) return baseTools;

  if (restrictions.disallowedTools) {
    return baseTools.filter(tool =>
      !restrictions.disallowedTools.includes(tool.name)
    );
  }

  if (restrictions.allowedTools) {
    return baseTools.filter(tool =>
      restrictions.allowedTools.includes(tool.name)
    );
  }

  return baseTools;
}

class MCPClient {
  private connection: ClientConnection | null = null;

  async connect(config: McpServerConfig) {
    const transport = await createTransport(config);

    this.connection = new ClientConnection(transport);

    // 等待服务器就绪
    await this.connection.ready();

    // 获取可用工具和资源
    const tools = await this.connection.listTools();
    const resources = await this.connection.listResources();

    return { tools, resources };
  }

  async callTool(name: string, input: Record<string, unknown>) {
    if (!this.connection) {
      throw new Error('MCP client not connected');
    }

    const result = await this.connection.callTool({ name, arguments: input });

    // 处理 MCP 响应
    return this.parseMcpResult(result);
  }
}

function createMcpTool(serverName: string, toolDef: MCPToolDefinition): Tool {
  return buildTool({
    name: `mcp__${serverName}__${toolDef.name}`,
    description: toolDef.description,
    inputJSONSchema: toolDef.inputSchema,
    isMcp: true,
    mcpInfo: { serverName, toolName: toolDef.name },

    async call(input, context) {
      const client = context.mcpClients.find(c => c.name === serverName);
      if (!client) {
        return { error: `MCP server "${serverName}" not found` };
      }

      return await client.callTool(toolDef.name, input);

    // MCP 工具默认只读，除非声明为 destructive
    isReadOnly: () => !toolDef.annotations?.destructive,
    isDestructive: () => !!toolDef.annotations?.destructive,
  });
}

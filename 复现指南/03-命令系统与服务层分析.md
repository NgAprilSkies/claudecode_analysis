const commitCommand: Command = {
  type: 'action',
  name: 'commit',
  description: '创建 git commit',
  source: 'builtin',

  async execute(args, context) {
    const { setToolJSX, getAppState } = context;

    // 1. 获取 git 状态
    const status = await runGitCommand(['status', '--porcelain']);
    const diff = await runGitCommand(['diff', 'HEAD']);

    // 2. 分析变更内容
    const analysis = await analyzeChanges(diff);

    // 3. 生成提交消息
    const commitMessage = await generateCommitMessage(analysis);

    // 4. 显示预览 UI
    setToolJSX({
      jsx: <CommitPreview
        changes={analysis}
        message={commitMessage}
      />,
      shouldHidePromptInput: true,
    });

    // 5. 等待用户确认
    const confirmed = await waitForUserConfirmation();

    if (!confirmed) {
      return { status: 'cancelled' };
    }

    // 6. 执行提交
    await runGitCommand(['add', '-A']);
    await runGitCommand(['commit', '-m', commitMessage]);

    // 7. 返回结果
    return {
      status: 'success',
      commitHash: await getCurrentCommitHash(),
      message: commitMessage,
    };
  },

const compactCommand: Command = {
  type: 'action',
  name: 'compact',
  description: '压缩对话上下文',
  source: 'builtin',

  async execute(args, context) {
    const { messages, setAppState } = context;

    // 1. 检查是否需要压缩
    const tokenCount = await countTokens(messages);
    const contextWindow = getContextWindowForModel(context.model);

    if (tokenCount < contextWindow * 0.8) {
      return { status: 'not_needed', tokenCount };
    }

    // 2. 查找压缩边界
    const boundaryIndex = findCompactBoundary(messages);

    // 3. 生成压缩摘要
    const messagesToCompact = messages.slice(0, boundaryIndex);
    const summary = await generateSummary(messagesToCompact);

    // 4. 构建压缩后的消息
    const compactedMessages = [
      createSystemMessage(`之前的对话摘要：${summary}`),
      ...messages.slice(boundaryIndex),
    ];

    // 5. 更新状态
    setAppState(prev => ({
      ...prev,
      messages: compactedMessages,
      compactedTokenCount: tokenCount - countTokens(messagesToCompact),
    }));

    return {
      status: 'success',
      originalTokens: tokenCount,
      compactedTokens: countTokens(compactedMessages),
      summary,
    };
  },
};

        const task = await updateTask(taskId, parseUpdates(updates), context);
        setAppState(prev => ({
          ...prev,
          tasks: prev.tasks.map(t => t.id === taskId ? task : t),
        }));

        const state = getAppState();
        const filtered = filterTasks(state.tasks, parseFilters(taskArgs));

      case 'delete': {
        const taskId = taskArgs[0];
        setAppState(prev => ({
          ...prev,
          tasks: prev.tasks.filter(t => t.id !== taskId),
        }));
        return { status: 'deleted', taskId };
      }
    }

class ClaudeAPIClient {
  private client: Anthropic;
  private apiKey: string;

  constructor(apiKey: string, options?: Anthropic.ClientOptions) {
    this.apiKey = apiKey;
    this.client = new Anthropic(options);

  async *streamMessage(params: MessageCreateParams): AsyncGenerator<MessageStreamEvent> {
    const {
      model,
      messages,
      system,
      max_tokens,
      tools,
      thinking,
    } = params;

    // 构建 API 请求
    const requestParams: Anthropic.MessageCreateParams = {

      max_tokens,

      system: normalizeSystemPrompt(system),
      stream: true,
      tools: tools ? normalizeTools(tools) : undefined,
      thinking: thinking, // 思考模式配置

    const stream = await this.client.messages.create(requestParams);

    // 流式处理响应
    for await (const event of stream) {
      yield this.normalizeEvent(event);

      // 累积使用量
      if (event.type === 'message_delta') {
        accumulateUsage(event.usage);
      }
    }
  }

  private normalizeEvent(event: Anthropic.MessageStreamEvent): MessageStreamEvent {
    // 标准化事件格式

  }

interface Usage {
  inputTokens: number;
  outputTokens: number;
  cacheReadInputTokens: number;
  cacheWriteInputTokens: number;
}

const PRICING = {
  'claude-opus-4-6-20260320': {
    input: 15 / 1_000_000,
    output: 75 / 1_000_000,
    cache_read: 1.5 / 1_000_000,
    cache_write: 18.75 / 1_000_000,
  },
  'claude-sonnet-4-6-20260320': {
    input: 3 / 1_000_000,
    output: 15 / 1_000_000,
    cache_read: 0.3 / 1_000_000,
    cache_write: 3.75 / 1_000_000,
  },
};

class CostTracker {
  private sessionUsage: Usage = {
    inputTokens: 0,
    outputTokens: 0,
    cacheReadInputTokens: 0,
    cacheWriteInputTokens: 0,
  };

  accumulateUsage(usage: Anthropic.Usage) {
    this.sessionUsage.inputTokens += usage.input_tokens;
    this.sessionUsage.outputTokens += usage.output_tokens;
    this.sessionUsage.cacheReadInputTokens += usage.cache_read_input_tokens || 0;
    this.sessionUsage.cacheWriteInputTokens += usage.cache_write_input_tokens || 0;
  }

  calculateCost(model: string): number {
    const pricing = PRICING[model as keyof typeof PRICING];
    if (!pricing) return 0;

    return (
      this.sessionUsage.inputTokens * pricing.input +
      this.sessionUsage.outputTokens * pricing.output +
      this.sessionUsage.cacheReadInputTokens * pricing.cache_read +
      this.sessionUsage.cacheWriteInputTokens * pricing.cache_write
    );
  }

  getSessionCost(): number {
    const model = getCurrentModel();
    return this.calculateCost(model);
  }
}

interface MCPServerConnection {
  name: string;
  config: McpServerConfig;
  client: MCPClient;
  tools: Tool[];
  resources: ServerResource[];
  status: 'connecting' | 'connected' | 'error';
  error?: string;
}

class MCPManager {
  private connections: Map<string, MCPServerConnection> = new Map();

  async connect(config: McpServerConfig): Promise<MCPServerConnection> {
    const { name, type, transport } = config;

    // 创建传输层
    const mcpTransport = await this.createTransport(type, transport);

    // 创建客户端
    const client = new MCPClient(mcpTransport);
    await client.ready();

    // 获取工具和资源
    const tools = await client.listTools();
    const resources = await client.listResources();

    // 包装为内部 Tool 格式
    const wrappedTools = tools.map(toolDef =>
      createMcpTool(name, toolDef)
    );

    const connection: MCPServerConnection = {
      name,
      config,
      client,
      tools: wrappedTools,
      resources,
      status: 'connected',
    };

    this.connections.set(name, connection);
    return connection;
  }

  private async createTransport(
    type: 'stdio' | 'sse' | 'websocket',
    config: TransportConfig
  ): Promise<Transport> {
    switch (type) {
      case 'stdio':
        return new StdioClientTransport({
          command: config.command,
          args: config.args,
          env: config.env,
        });

      case 'sse':
        return new SSEClientTransport(new URL(config.url));

      case 'websocket':
        return new WebSocketClientTransport(config.url);
    }
  }

  async disconnect(name: string) {
    const connection = this.connections.get(name);
    if (connection) {
      await connection.client.close();
      this.connections.delete(name);
    }
  }

  getAllTools(): Tool[] {
    return Array.from(this.connections.values())
      .flatMap(conn => conn.tools);
  }
}

interface Plugin {
  name: string;
  version: string;
  description: string;
  commands?: Command[];
  skills?: Skill[];
  hooks?: Hook[];
  tools?: Tool[];
}

class PluginLoader {
  private plugins: Map<string, Plugin> = new Map();

  async loadAllPlugins(): Promise<{ enabled: string[]; errors: Error[] }> {
    const enabled: string[] = [];
    const errors: Error[] = [];

    // 1. 获取插件目录
    const pluginDirs = getPluginDirectories();

    // 2. 并行加载所有插件
    const loadPromises = pluginDirs.map(async (dir) => {
      try {
        const plugin = await this.loadPlugin(dir);
        if (plugin) {
          this.plugins.set(plugin.name, plugin);
          enabled.push(plugin.name);
        }
      } catch (error) {
        errors.push(new Error(`加载插件 ${dir} 失败：${error}`));
      }
    });

    await Promise.all(loadPromises);

    return { enabled, errors };
  }

  private async loadPlugin(dir: string): Promise<Plugin | null> {
    // 读取 package.json
    const pkgPath = path.join(dir, 'package.json');
    const pkg = JSON.parse(await readFile(pkgPath, 'utf-8'));

    // 读取入口文件
    const entryPoint = pkg.claude_plugin?.main || pkg.main;
    if (!entryPoint) return null;

    const pluginModule = await import(path.join(dir, entryPoint));

    // 验证插件结构
    if (!this.validatePlugin(pluginModule)) {
      return null;
    }

    return {
      name: pkg.name,
      version: pkg.version,
      description: pkg.description,
      commands: pluginModule.commands || [],
      skills: pluginModule.skills || [],
      hooks: pluginModule.hooks || [],
      tools: pluginModule.tools || [],
    };
  }

  private validatePlugin(module: unknown): boolean {
    // 验证必需的导出
    return typeof module === 'object' && module !== null;
  }

  getPluginCommands(): Command[] {
    return Array.from(this.plugins.values())
      .flatMap(plugin => plugin.commands || []);
  }

  getPluginSkills(): Skill[] {
    return Array.from(this.plugins.values())
      .flatMap(plugin => plugin.skills || []);
  }
}

        trigger: skillDef.trigger,
        prompt: prompt,
        source: 'dynamic',
      });
    } catch {

    }
  }

}

function matchSkill(input: string, skills: Skill[]): Skill | null {
  for (const skill of skills) {
    for (const trigger of skill.trigger) {
      if (input.toLowerCase().includes(trigger.toLowerCase())) {
        return skill;
      }
    }
  }
  return null;

// 创建存储
export function createStore(initialState: AppState) {
  let state = initialState;
  const listeners: Set<(state: AppState) => void> = new Set();

  return {
    getState: () => state,

    setState: (updater: (prev: AppState) => AppState) => {
      state = updater(state);
      listeners.forEach(listener => listener(state));
    },

    subscribe: (listener: (state: AppState) => void) => {
      listeners.add(listener);
      return () => listeners.delete(listener);
    },
  };
}

interface SessionRecord {
  id: string;
  createdAt: number;
  updatedAt: number;
  messages: Message[];
  cost: number;
  model: string;
  cwd: string;
  title?: string;
}

const SESSION_STORAGE_DIR = path.join(os.homedir(), '.claude', 'sessions');

async function saveSession(session: SessionRecord) {
  const sessionFile = path.join(SESSION_STORAGE_DIR, `${session.id}.json`);
  await writeFile(sessionFile, JSON.stringify(session, null, 2));
}

async function loadSession(sessionId: string): Promise<SessionRecord | null> {
  const sessionFile = path.join(SESSION_STORAGE_DIR, `${sessionId}.json`);
  try {
    const content = await readFile(sessionFile, 'utf-8');
    return JSON.parse(content);
  } catch {
    return null;
  }
}

async function listSessions(): Promise<SessionRecord[]> {
  const files = await readdir(SESSION_STORAGE_DIR);
  const sessions = await Promise.all(
    files
      .filter(f => f.endsWith('.json'))
      .map(async (file) => {
        const content = await readFile(path.join(SESSION_STORAGE_DIR, file), 'utf-8');
        return JSON.parse(content);
      })
  );

  // 按更新时间排序
  return sessions.sort((a, b) => b.updatedAt - a.updatedAt);
}
